    Full-screen shared component. No props — reads context from hooks:
    ```typescript
    import { useRoutePrefix } from '../hooks/useRoutePrefix';

    import ScreenHeader from '../components/ScreenHeader';
    import CatalogPlantationCard from '../components/CatalogPlantationCard';
    import DownloadProgressModal from '../components/DownloadProgressModal';
    import { colors, fontSize, spacing, borderRadius, fonts } from '../theme';
    import Ionicons from '@expo/vector-icons/Ionicons';
    ```

    State variables:
    - catalogItems: ServerPlantation[] (initially [])
    - localIds: Set<string> (initially new Set())
    - selectedIds: Set<string> (initially new Set())
    - loadingCatalog: boolean (initially true)
    - catalogError: string | null (initially null)
    - downloadState: 'idle' | 'downloading' | 'done' (initially 'idle')
    - downloadProgress: DownloadProgress | null (initially null)
    - downloadResults: DownloadResult[] (initially [])

    Derived values:
    - isAdmin = routePrefix === '(admin)'
    - organizacionId = profile?.organizacionId ?? ''

    useEffect on mount (deps: [isOnline, profile]):
    - If !isOnline: setCatalogError('No se pudo cargar el catalogo'); setLoadingCatalog(false); return;
    - If !profile?.organizacionId: return; (wait for profile to load)
    - Call loadCatalog()

    async function loadCatalog():
    - setLoadingCatalog(true); setCatalogError(null);
    - try { const [items, ids] = await Promise.all([getServerCatalog(isAdmin, userId, organizacionId), getLocalPlantationIds()]); setCatalogItems(items); setLocalIds(ids); }
    - catch (e) { setCatalogError('No se pudo cargar el catalogo'); }
    - finally { setLoadingCatalog(false); }

    Toggle handler: toggleSelection(id: string) — add/remove from selectedIds Set (create new Set for state update)

    Download handler: handleBatchDownload():
    - Build selected array from catalogItems filtered by selectedIds
    - setDownloadState('downloading'); setDownloadProgress(null); setDownloadResults([]);
    - try { const results = await batchDownload(selected, (p) => setDownloadProgress(p)); setDownloadResults(results); }
    - catch (e) { setDownloadResults(selected.map(s => ({ success: false, id: s.id, nombre: s.lugar }))); }
    - finally { setDownloadState('done'); }

    Dismiss handler: handleDismiss():
    - setDownloadState('idle'); setSelectedIds(new Set());
    - Refresh local IDs: getLocalPlantationIds().then(ids => setLocalIds(ids));

    Render:
    - SafeAreaView with backgroundColor colors.background, flex 1
    - ScreenHeader title="Catalogo de plantaciones"
    - Loading state: centered ActivityIndicator color={colors.primary} size="large"
    - Error state: centered Ionicons "cloud-offline-outline" size=48 color={colors.textMuted}, Text "No se pudo cargar el catalogo" (emptyTitle style), Text "Verifica tu conexion y vuelve a intentarlo" (emptySubtext style), Pressable "Reintentar carga" (onPress: loadCatalog)
    - Empty state (catalogItems.length === 0 after loading): Ionicons "leaf-outline" size=48, Text "No hay plantaciones disponibles", Text body = isAdmin ? "Todas las plantaciones del servidor ya estan en tu dispositivo" : "No tenes plantaciones asignadas en el servidor"
    - Loaded state: FlatList with contentContainerStyle { paddingHorizontal: spacing.xxl, paddingTop: spacing.xl, gap: spacing.xl, paddingBottom: 96 }

    - DownloadProgressModal state={downloadState} progress={downloadProgress} results={downloadResults} onDismiss={handleDismiss}
