Rules from CONTEXT.md (locked decisions):

```tsx
import { Pressable, Text, StyleSheet, View } from 'react-native';
import { useState } from 'react';

interface Props {
  codigo: string;
  nombre: string;
  onPress: () => void;
  isNN?: boolean;  // true for the N/N special button
  disabled?: boolean;
}

export default function SpeciesButton({ codigo, nombre, onPress, isNN = false, disabled = false }: Props) {
  const [pressed, setPressed] = useState(false);

  return (
    <Pressable
      onPressIn={() => setPressed(true)}
      onPressOut={() => setPressed(false)}
      onPress={onPress}
      disabled={disabled}
      style={[
        styles.button,
        isNN && styles.buttonNN,
        pressed && (isNN ? styles.buttonNNPressed : styles.buttonPressed),
        disabled && styles.buttonDisabled,
      ]}
    >
      <Text style={[styles.code, isNN && styles.codeNN]}>{codigo}</Text>
      <Text style={[styles.name, isNN && styles.nameNN]}>{nombre}</Text>
    </Pressable>
  );
}
```

Step 3 — Create mobile/src/components/TreeRow.tsx:

Shows one row in the last-3 list. Position number + species code (or "N/N"). Tapping deletes this tree ONLY IF it is the last tree (matching what deleteLastTree does). The screen passes isLast={true} for the first item in the lastThree array (which is the highest posicion due to DESC order).

```tsx
import { Pressable, Text, View, StyleSheet } from 'react-native';

interface Props {
  posicion: number;
  especieCodigo: string | null;  // null = N/N
  subId: string;
  isLast: boolean;  // only last tree can be deleted (undo)
  onDelete?: () => void;
}

export default function TreeRow({ posicion, especieCodigo, subId, isLast, onDelete }: Props) {
  const displayCode = especieCodigo ?? 'N/N';

  return (
    <View style={[styles.row, isLast && styles.rowLast]}>
      <Text style={styles.position}>{posicion}</Text>
      <Text style={styles.code}>{displayCode}</Text>
      <Text style={styles.subId} numberOfLines={1}>{subId}</Text>
      {isLast && onDelete && (
        <Pressable onPress={onDelete} style={styles.undoButton} hitSlop={12}>
          <Text style={styles.undoText}>Deshacer</Text>
        </Pressable>
      )}
    </View>
  );
}

const styles = StyleSheet.create({
  row: {
    flexDirection: 'row',
    alignItems: 'center',
    paddingVertical: 8,
    paddingHorizontal: 12,
    backgroundColor: '#f9f9f9',
    borderRadius: 6,
    gap: 8,
  },
  rowLast: {
    backgroundColor: '#f0f7f0',
    borderLeftWidth: 3,
    borderLeftColor: '#4caf50',
  },
  position: { fontSize: 16, fontWeight: 'bold', color: '#333', minWidth: 28 },
  code: { fontSize: 16, fontWeight: '600', color: '#2d6a2d', minWidth: 48 },
  subId: { fontSize: 12, color: '#888', flex: 1 },
  undoButton: { paddingHorizontal: 8, paddingVertical: 4, backgroundColor: '#ffebee', borderRadius: 4 },
  undoText: { fontSize: 12, color: '#c62828', fontWeight: '600' },
});
```

NOTE: TreeRow receives `especieCodigo` but the trees schema only stores `especieId`. The registration screen needs to join or resolve the species code when building the last-3 display. Options: (a) store codigo in tree row (denormalized), or (b) join species when querying for display. Option (b) is architecturally clean. The useTrees hook returns `{ especieId }` — the registration screen must either enrich the data with species codes or show only position + subId in the last-3 (which already encodes the species code). Simplest correct approach: display subId in the last-3 row (it encodes code+position), and show position + subId. The TreeRow component above handles this. `especieCodigo` can be extracted from subId if needed or passed separately.
