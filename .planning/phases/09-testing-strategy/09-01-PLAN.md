4. Create `tests/helpers/integrationDb.ts`:
   ```typescript
   import Database from 'better-sqlite3';
   import { drizzle } from 'drizzle-orm/better-sqlite3';
   import * as schema from '../../src/database/schema';
   import path from 'path';

   export function createTestDb() {
     const sqlite = new Database(':memory:');
     // Read and execute migration SQL files
     const migrationsDir = path.join(__dirname, '../../drizzle');
     // Use drizzle-orm/better-sqlite3/migrator migrate() function
     const db = drizzle(sqlite, { schema });
     // Run migrations using migrate() from drizzle-orm/better-sqlite3/migrator
     return { db, sqlite };
   }

   export function closeTestDb(sqlite: ReturnType<typeof Database>) {
     sqlite.close();
   }
   ```
   Use `path.join(__dirname, '../../drizzle')` for migration folder path (per Pitfall 1 from research). Import `migrate` from `drizzle-orm/better-sqlite3/migrator`.
